数据结构与算法「线性表c」

代码区块

  1. * 结构代码
  2. #define LIST_INIT_SIZE 20   //存储空间初始分配量
  3. #define LIST_INCREMENT 2    //存储空间分配增量
  4. typedef int ElemType;       //根据实际类型而定，假设为int
  5. typedef struct {
  6.     ElemType *elem;         //数据存储空间基址，或者(ElemType elem[LIST_INIT_SIZE])
  7.     int size;               //当前数据存储空间容量
  8.     int len;                //当前数据长度(元素个数)
  9. } SQList;                   //顺序存储结构的结构体定义
  11. * 构造空表
  12.     - 1. 分配初始存储空间
  13.     - 2. 如果分配失败，抛出异常
  14.     - 3. 初始化当前数据存储容量和长度
  15. void InitSequenceList(SQList *list) {
  16.     list->elem = malloc(sizeof(ElemType) * LIST_INIT_SIZE);
  17.     if (!list->elem) {
  18.         exit(0);
  19.     }
  20.     list->size = LIST_INIT_SIZE;
  21.     list->len = 0;
  22. }
  24. * 获得元素
  25.     - 1. 如果查找的位置不合理，抛出异常
  26.     - 2. 根据基址及查找位置，通过地址偏移获得元素
  27. void GetSequenceListElement(SQList *list, int index, ElemType *e) {
  28.     if (index <= 0 || index > list->len) {
  29.         exit(-1);
  30.     }
  31.     if (0 == list->len) {
  32.         exit(0);
  33.     }
  34.     *e = *(list->elem + (index-1));
  35. }
  37. * 插入元素
  38.     - 1. 如果插入的位置不合理，抛出异常
  39.     - 2. 如果当前存储空间容量不足，则抛出异常或动态增加存储容量
  40.     - 3. 从最后一个元素开始向前遍历至第i个位置，并将其间元素都向后移动一个位置
  41.     - 4. 将元素插入位置i处，并将表长度+1
  42. void InsertSequenceListElement(SQList *list, int index, ElemType elem) {
  43.     if (index <= 0 || index > list->len) {
  44.         exit(-1);
  45.     }
  46.     if (list->len == list->size) {
  47.         //exit(0);
  48.         ElemType *n_elem = realloc(list->elem, sizeof(ElemType) * (list->size+LIST_INCREMENT));
  49.         if (!n_elem) {
  50.             exit(0);
  51.         }
  52.         list->elem = n_elem;
  53.         list->size += LIST_INCREMENT;
  54.     }
  55.     ElemType *loc = list->elem + (index-1);
  56.     ElemType *tail = list->elem + list->len;
  57.     for (; loc <= tail; loc++) {
  58.         *(loc+1) = *loc;
  59.     }
  60.     *loc = elem;
  61.     list->len += 1;
  62. }
  64. * 删除元素
  65.     - 1. 如果删除的位置不合理，抛出异常
  66.     - 2. 从第i个位置开始向后遍历至最后一个元素，并将其间元素都向前移动一个位置
  67.     - 3. 将表长度-1
  68. void DeleteSequenceListElement(SQList *list, int index, ElemType *e) {
  69.     if (index <= 0 || index > list->len) {
  70.         exit(-1);
  71.     }
  72.     ElemType *loc = list->elem + (index-1);
  73.     ElemType *tail = list->elem + list->len;
  74.     *e = *loc;
  75.     for (; loc < tail; loc++) {
  76.         *loc = *(loc+1);
  77.     }
  78.     list->len -= 1;
  79. }
  81. * 查找元素
  82.     - 1. 从第1个位置开始向后遍历元素，当找到元素时则退出
  83.     - 2. 如果遍历过线性表长度或元素不存在，抛出异常
  84. int LocateSequenceListElement(SQList *list, ElemType elem) {
  85.     int i = 0;
  86.     while (*(list->elem+i) != elem && i < list->len) {
  87.         i++;
  88.     }
  89.     if (i >= list->len) {
  90.         return -1;
  91.     }
  92.     return i+1;
  93. }
  95. * 表的长度
  96.     - 1. 线性表结构的当前数据长度
  97. int SequenceListLength(SQList *list) {
  98.     return list->len;
  99.  }
  102. * 表的判空
  103.     - 1. 线性表长度是否为0，则表示表空
  104. bool IsSequenceListEmpty(SQList *list) {
  105.     return (0 == list->len);
  106. }
  108. * 表的置空
  109.     - 1. 线性表长度置为0，则表示表空
  110. void ClearSequenceList(SQList *list) {
  111.     list->len = 0;
  112. }
  114. * 表的销毁
  115.     - 1. 释放结构体所占内存空间
  116.     - 2. 置空结构体初始化变量
  117. void DestroySequenceList(SQList *list) {
  118.     free(list->elem);
  119.     list->elem = NULL;
  120.     list->size = 0;
  121.     list->len = 0;
  122. }
